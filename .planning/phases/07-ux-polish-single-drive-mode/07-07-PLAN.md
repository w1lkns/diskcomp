   ```python
   def show_action_menu():
       """
       Display post-scan action menu (D-23).

       Menu options:
         1) Review and delete interactively (per-file confirmation)
         2) Batch delete (dry-run + type DELETE to confirm)
         3) Exit (report saved, no deletion)

       Returns:
           One of: 'interactive', 'batch', 'exit'
       """
       while True:
           print("""
What next?
  1) Review and delete interactively
  2) Batch delete (preview + confirm)
  3) Exit (report saved)
           """.strip())

           choice = input("Select (1-3): ").strip()

           if choice in ['1', '2', '3']:
               mapping = {'1': 'interactive', '2': 'batch', '3': 'exit'}
               return mapping[choice]
           else:
               print("Invalid choice. Please enter 1, 2, or 3.")
               print()
   ```

2. In main(), after show_plain_language_summary() and show_next_steps(), add action menu routing:
   ```python
   # Show action menu only if duplicates found (D-24)
   if summary_dict.get('duplicate_count', 0) > 0:
       action = show_action_menu()

       if action == 'interactive' or action == 'batch':
           # Load candidates from report and launch deletion workflow
           reader = ReportReader.load(writer.output_path)
           candidates = reader.candidates  # or reader.get_candidates()
           mode = 'interactive' if action == 'interactive' else 'batch'

           orchestrator = DeletionOrchestrator(ui=ui)
           result = (
               orchestrator.interactive_mode(candidates)
               if mode == 'interactive'
               else orchestrator.batch_mode(candidates)
           )

           # Display result summary
           ui.ok(f"Deletion complete: {result.files_deleted} deleted, {result.space_freed_mb:.2f} MB freed")
           if result.errors:
               for error in result.errors:
                   ui.error(f"Error: {error}")

           # Show undo log path
           if result.undo_log_path:
               ui.ok(f"Undo log: {result.undo_log_path}")

           ui.close()
           return 0

       elif action == 'exit':
           # Already shown next steps above, just exit
           ui.close()
           return 0
   else:
       # No duplicates, exit cleanly (D-24)
       ui.close()
       return 0
   ```

   Note: This presumes ui is available (from the scan setup). Ensure ui.close() is only called once.
