2. **main() function** — orchestrate the full pipeline:
   ```python
   def main(args=None):
       # Parse arguments (allow passing args for testing)
       args = parse_args(args)

       # Validate paths exist and are readable
       for path in [args.keep, args.other]:
           if not os.path.isdir(path):
               print(f"Error: {path} is not a readable directory")
               return 1

       # Scan both drives
       keep_scanner = FileScanner(args.keep)
       other_scanner = FileScanner(args.other)

       keep_result = keep_scanner.scan(dry_run=args.dry_run, limit=args.limit)
       other_result = other_scanner.scan(dry_run=args.dry_run, limit=args.limit)

       print(f"Keep:  {keep_result.file_count} files, {keep_result.total_size_bytes} bytes")
       print(f"Other: {other_result.file_count} files, {other_result.total_size_bytes} bytes")

       # If dry-run, skip hashing and exit
       if args.dry_run:
           print("(dry-run mode: hashing skipped)")
           return 0

       # Hash all files
       hasher = FileHasher()
       keep_result.files = [hasher.hash_file_record(f) for f in keep_result.files]
       other_result.files = [hasher.hash_file_record(f) for f in other_result.files]

       # Classify duplicates
       classifier = DuplicateClassifier(keep_result, other_result)
       classification = classifier.classify()

       # Write report
       writer = ReportWriter(output_path=args.output)
       if args.format == 'json':
           writer.write_json(classification)
       else:  # default csv
           writer.write_csv(classification)

       print(f"Report written to: {writer.output_path}")
       return 0
   ```

import tempfile
import os
from pathlib import Path
from diskcomp.scanner import FileScanner, NOISE_PATTERNS

class TestFileScanner(unittest.TestCase):
    def setUp(self):
        # Create temporary test directory with files
        self.test_dir = tempfile.mkdtemp()
        # Create test files
        Path(self.test_dir, "file1.txt").write_text("content1")
        Path(self.test_dir, "file2.txt").write_text("content2")
        Path(self.test_dir, ".DS_Store").write_text("noise")  # noise file
        Path(self.test_dir, "subdir").mkdir()
        Path(self.test_dir, "subdir", "file3.txt").write_text("content3")

    def tearDown(self):
        # Clean up test directory
        import shutil
        shutil.rmtree(self.test_dir)

    def test_scan_counts_files(self):
        scanner = FileScanner(self.test_dir)
        result = scanner.scan()
        assert result.file_count > 0
        assert result.total_size_bytes > 0

    def test_scan_skips_noise(self):
        scanner = FileScanner(self.test_dir)
        result = scanner.scan()
        assert result.skipped_noise_count >= 1  # .DS_Store should be skipped
        # Verify .DS_Store is not in result.files
        file_names = [f.path for f in result.files]
        assert not any(".DS_Store" in f for f in file_names)

    def test_scan_filters_small_files(self):
        # Create file < 1KB
        tiny_file = Path(self.test_dir, "tiny.txt")
        tiny_file.write_text("x" * 100)  # 100 bytes
        scanner = FileScanner(self.test_dir, min_size_bytes=1024)
        result = scanner.scan()
        # tiny.txt should not be in result (too small)
        file_names = [f.path for f in result.files]
        assert not any("tiny.txt" in f for f in file_names)

    def test_dry_run_skips_hashing(self):
        scanner = FileScanner(self.test_dir)
        result = scanner.scan(dry_run=True)
        # In dry-run, files may not be fully read
        assert result.file_count > 0  # But file count should still work

    def test_scan_respects_limit(self):
        scanner = FileScanner(self.test_dir)
        result = scanner.scan(limit=1)
        assert len(result.files) <= 1

**tests/test_hasher.py** — test FileHasher:
```python
import unittest
import tempfile
from pathlib import Path
from diskcomp.hasher import FileHasher
from diskcomp.types import FileNotReadableError

class TestFileHasher(unittest.TestCase):
    def setUp(self):
        self.hasher = FileHasher()
        self.test_file = tempfile.NamedTemporaryFile(delete=False)
        self.test_file.write(b"test content")
        self.test_file.close()

    def tearDown(self):
        import os
        os.unlink(self.test_file.name)

    def test_hash_returns_hex_string(self):
        hash_val = self.hasher.hash_file(self.test_file.name)
        assert isinstance(hash_val, str)
        assert len(hash_val) == 64  # SHA256 hex = 64 chars
        assert all(c in "0123456789abcdef" for c in hash_val)

    def test_same_file_same_hash(self):
        hash1 = self.hasher.hash_file(self.test_file.name)
        hash2 = self.hasher.hash_file(self.test_file.name)
        assert hash1 == hash2

    def test_different_files_different_hashes(self):
        test_file2 = tempfile.NamedTemporaryFile(delete=False)
        test_file2.write(b"different content")
        test_file2.close()
        try:
            hash1 = self.hasher.hash_file(self.test_file.name)
            hash2 = self.hasher.hash_file(test_file2.name)
            assert hash1 != hash2
        finally:
            import os
            os.unlink(test_file2.name)

    def test_nonexistent_file_raises_error(self):
        with self.assertRaises(FileNotReadableError):
            self.hasher.hash_file("/nonexistent/file.txt")

if __name__ == "__main__":
    unittest.main()
```

import unittest
import tempfile
from pathlib import Path
from diskcomp.scanner import FileScanner
from diskcomp.hasher import FileHasher
from diskcomp.reporter import DuplicateClassifier, ReportWriter

class TestIntegration(unittest.TestCase):
    def test_full_pipeline(self):
        # Create two test directories with files
        keep_dir = tempfile.mkdtemp()
        other_dir = tempfile.mkdtemp()

        try:
            # Create identical files
            keep_file = Path(keep_dir, "file.txt")
            keep_file.write_text("same content")

            other_file = Path(other_dir, "file.txt")
            other_file.write_text("same content")

            # Create unique file
            unique_keep = Path(keep_dir, "unique.txt")
            unique_keep.write_text("unique to keep")

            # Scan both
            keep_scanner = FileScanner(keep_dir)
            other_scanner = FileScanner(other_dir)

            keep_result = keep_scanner.scan()
            other_result = other_scanner.scan()

            # Hash files
            hasher = FileHasher()
            keep_result.files = [hasher.hash_file_record(f) for f in keep_result.files]
            other_result.files = [hasher.hash_file_record(f) for f in other_result.files]

            # Classify
            classifier = DuplicateClassifier(keep_result, other_result)
            classification = classifier.classify()

            # Write report
            report_dir = tempfile.mkdtemp()
            try:
                writer = ReportWriter(output_dir=report_dir)
                writer.write_csv(classification)

                # Verify report exists
                assert Path(writer.output_path).exists()
                # Verify report has duplicate entry
                assert len(classification['duplicates']) >= 1
            finally:
                import shutil
                shutil.rmtree(report_dir)
        finally:
            import shutil
            shutil.rmtree(keep_dir)
            shutil.rmtree(other_dir)

if __name__ == "__main__":
    unittest.main()
